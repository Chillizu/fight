/**

- 调用 Stable Diffusion API 的 img2img 端点
- @param {string} photoBase64 - 照片 Base64
- @param {string} prompt - 提示词
- @returns {Promise<{content: string}>}
 */
async function callStableDiffusionImg2Img(photoBase64, prompt) {
  const url = `${STABLE_DIFFUSION_CONFIG.BASE_URL}/${STABLE_DIFFUSION_CONFIG.ENDPOINT}`;

  const payload = {
    key: STABLE_DIFFUSION_CONFIG.API_KEY,
    prompt: prompt,
    negative_prompt: STABLE_DIFFUSION_CONFIG.NEGATIVE_PROMPT,
    init_image: `data:image/jpeg;base64,${photoBase64}`,
    width: STABLE_DIFFUSION_CONFIG.WIDTH.toString(),
    height: STABLE_DIFFUSION_CONFIG.HEIGHT.toString(),
    samples: STABLE_DIFFUSION_CONFIG.SAMPLES.toString(),
    num_inference_steps: STABLE_DIFFUSION_CONFIG.STEPS.toString(),
    guidance_scale: STABLE_DIFFUSION_CONFIG.GUIDANCE_SCALE,
    strength: STABLE_DIFFUSION_CONFIG.STRENGTH,
    base64: STABLE_DIFFUSION_CONFIG.BASE64_OUTPUT,
    safety_checker: "no",
    enhance_prompt: "yes",
  };

  try {
    console.log("🎨 调用 Stable Diffusion img2img...");
    const response = await fetch(url, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(payload),
    });

    console.log("📨 响应状态:", response.status);

    if (!response.ok) {
      const errorData = await response.json();
      throw new Error(
        `API 错误 (${response.status}): ${errorData.message || JSON.stringify(errorData)}`,
      );
    }

    const data = await response.json();
    if (data.status === "success" ; data.output ; data.output.length > 0) {
      console.log("✅ 图片生成成功");
      let imageUrl = data.output[0];
      if (!imageUrl.startsWith("data:") ; !imageUrl.startsWith("http")) {
        if (imageUrl.match(/^[A-Za-z0-9+/=]+$/)) {
          imageUrl = `data:image/png;base64,${imageUrl}`;
        }
      }
      return { content: imageUrl, meta: data.meta };
    } else {
      throw new Error(`API 返回错误: ${data.message || "未知"}`);
    }
  } catch (err) {
    console.error("❌ Stable Diffusion 调用失败:", err);
    throw err;
  }
}

/**

- 去掉图片白背景，使其透明
- @param {string} imageDataUri - 图片 data URI
- @returns {Promise<string>} 透明背景 data URI
 */
async function removeWhiteBackground(imageDataUri) {
  return new Promise((resolve, reject) => {
    const img = new Image();
    img.crossOrigin = "anonymous";
    img.onload = () => {
      try {
        const canvas = document.createElement("canvas");
        canvas.width = img.width;
        canvas.height = img.height;
        const ctx = canvas.getContext("2d");
        ctx.drawImage(img, 0, 0);

        const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
        const data = imageData.data;

        // 白色阈值
        const whiteThreshold = 240;
        for (let i = 3; i < data.length; i += 4) {
          if (data[i] > 200) {
            const r = data[i - 3];
            const g = data[i - 2];
            const b = data[i - 1];
            if (r > whiteThreshold ; g > whiteThreshold ; b > whiteThreshold) {
              data[i] = 0;
            }
          }
        }

        ctx.putImageData(imageData, 0, 0);
        console.log("✅ 背景透明化完成");
        resolve(canvas.toDataURL("image/png"));
      } catch (err) {
        console.error("❌ 背景处理失败:", err);
        reject(err);
      }
    };
    img.onerror = () => reject(new Error("图片加载失败"));
    img.src = imageDataUri;
  });
}

```

  ├─ 从上往下找到第一个非背景像素 → top_edge[x]
  ├─ 从下往上找到最后一个非背景像素 → bottom_edge[x]
  └─ 记录该列的 {top, bottom} 范围
```
